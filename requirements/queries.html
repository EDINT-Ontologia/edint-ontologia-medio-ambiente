MED03: ¿De qué tipo es cada sensor? 
PREFIX edintmed: <http://vocab.linkeddata.es/datosabiertos/def/medioambiente/>
PREFIX edintST: <http://vocab.linkeddata.es/datosabiertos/kos/edint/sensortypes/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?sensor ?type ?typeLabel
WHERE {
  ?sensor a edintmed:EnvironmentalSensor ;
          edintmed:hasSensorType ?type .
  OPTIONAL { ?type rdfs:label ?typeLabel }
}
Variante: ¿Cuántos sensores de cada tipo? 
PREFIX edintmed: <http://vocab.linkeddata.es/datosabiertos/def/medioambiente/>
PREFIX edintST: <http://vocab.linkeddata.es/datosabiertos/kos/edint/sensortypes/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?sensorType ?sensorTypeLabel (COUNT(?sensor) AS ?numSensors)
WHERE {
  ?sensor a edintmed:EnvironmentalSensor ;
          edintmed:hasSensorType ?sensorType .

  OPTIONAL { ?sensorType rdfs:label ?sensorTypeLabel }
}
GROUP BY ?sensorType ?sensorTypeLabel
ORDER BY ?sensorTypeLabel

MED04 ¿Quién ha fabricado cada sensor? 
PREFIX schema: <http://schema.org/>
PREFIX edintmed: <http://vocab.linkeddata.es/datosabiertos/def/medioambiente/>

SELECT ?manufacturer ?manufacturerLabel ?sensor 
WHERE {
  ?sensor a edintmed:EnvironmentalSensor ;
          schema:manufacturer ?manufacturer .
  OPTIONAL { ?manufacturer rdfs:label ?manufacturerLabel }
}
ORDER BY ?manufacturer

MED05: ¿Qué observaciones ha generado cada sensor? 
PREFIX sosa: <http://www.w3.org/ns/sosa/>
PREFIX edintmed: <http://vocab.linkeddata.es/datosabiertos/def/medioambiente/>

SELECT ?sensor ?observation
WHERE {
  ?observation a edintmed:EnvironmentalObservation ;
               sosa:madeBySensor ?sensor .
}
Variante: Para un sensor concreto 
PREFIX edintmed: <http://vocab.linkeddata.es/datosabiertos/def/medioambiente/>
PREFIX sosa: <http://www.w3.org/ns/sosa/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?observation ?observedProperty ?result ?resultTime
WHERE {
  ?observation a edintmed:EnvironmentalObservation ;
               sosa:madeBySensor <URI-del-sensor> ;
               sosa:observedProperty ?observedProperty .

  OPTIONAL { ?observation sosa:hasSimpleResult ?result }
  OPTIONAL { ?observation sosa:resultTime ?resultTime }
}
ORDER BY ?resultTime

MED06: ¿Qué observaciones se han hecho para una propiedad concreta en un lugar concreto? 
PREFIX edintmed: <http://vocab.linkeddata.es/datosabiertos/def/medioambiente/>
PREFIX sosa: <http://www.w3.org/ns/sosa/>
PREFIX geo: <http://www.opengis.net/ont/geosparql#>

SELECT ?observation ?sensor ?platform ?time ?result
WHERE {
  # Observaciones de la propiedad concreta
  ?observation a edintmed:EnvironmentalObservation ;
               sosa:observedProperty <URI-de-la-propiedad> ;
               sosa:madeBySensor ?sensor .

  # Sensor es una Feature, que tiene una geometría
  ?sensor hasGeometry ?sensorgeo .
  
  # Geometría del lugar
  <URI-del-lugar> geo:hasGeometry ?areaGeom.

 # Relación espacial
  FILTER( geo:sfWithin(?sensorgeo, ?areaGeom) )

  OPTIONAL { ?observation sosa:resultTime ?time }
  OPTIONAL { ?observation sosa:hasSimpleResult ?result }
}
ORDER BY ?time
